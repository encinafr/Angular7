        function gotoStep(step, stateParams, optionParams)
        {
            for (var stepIndex = 0; stepIndex < ctrl.steps.length; stepIndex++) {
                var stepItem = ctrl.steps[stepIndex];
                if (stepItem.step == step) {
                    ctrl.currentStep = stepItem.step;
                    $state.go(stepItem.state, stateParams, optionParams);
                    break;
                }
            }

            if (ctrl.vm.schedule == 1) {
                 document.getElementById("horaReserva").max = '23:00';
            }
        }
